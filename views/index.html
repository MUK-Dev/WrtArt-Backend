WrtArt Api Information

Errors

All errors you will recieve from all endpoints

Response

          
              {
                "status": String,
                "message": String,
              }

ARTICLES

GET /articles

Get all articles

Response

          
              "status": Number,
              "message": String,
              "articles": [
                {
                  "_id": String,
                  "title": String,
                  "author": String,
                  "content": Editor JS Object which is needed to display formatted text,
                  "additions": [
                    {
                      "name": String,
                      "comment": String
                    }
                  ]
                }
              ]
POST /articles

Post new article

Request

           
          {
            "title": String,
            "author": String,
            "content": Editor JS Object which is needed to display formatted text,
          },
          Authorization: Bearer {token}

Response

          
            {
              "status": Number,
              "message": String
            }
GET /articles/{id}

Get one articles

Response

          
            {
              "status": Number,
              "message": String,
              "article": {
                "_id": String,
                "title": String,
                "author": String,
                "content": Editor JS Object which is needed to display formatted text,
                "additions": [
                  {
                    "name": String,
                    "comment": String
                  }
                ]
              }
            }
PATCH /articles/{id}

Patch existing article

Request

           
          {
            "name": String,
            "comment": String,
          },
          Authorization: Bearer {token}

Response

          
            {
              "status": Number,
              "message": String
            }

AUTHENTICATION

POST /login

Login existing user

Request

           
          {
            "email": String,
            "password": String
          }

Response

           
          {
            "status": Number,
            "message": String,
            "user-info": {
              "_id": String,
              "name": String,
              "email": String
            },
            "token": String
          }
POST /register

Register new user

Request

           
          {
            "email": String,
            "name": String,
            "password": String
          }

Response

           
          {
            "status": Number,
            "message": String,
            "user-info": {
              "_id": String,
              "name": String,
              "email": String
            },
            "token": String
          }
